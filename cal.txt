    package com.example.cal_vis

    import androidx.appcompat.app.AppCompatActivity
    import android.os.Bundle
    import kotlinx.android.synthetic.main.activity_main.*
    import java.lang.Math.*
    import kotlin.math.cos
    import kotlin.math.sin
    import kotlin.math.tan
    class MainActivity : AppCompatActivity() {
    var firstnumber : Float = 0F
    var secondnumber : Float = 0F
    var switch : Boolean = false
    var text:String = ""
    var operator:Char = '.'
    var dotflag:Boolean = false
    override fun onCreate(savedInstanceState: Bundle?) {
    super.onCreate(savedInstanceState)
    setContentView(R.layout.activity_main)
    one.setOnClickListener { append('1') }
    two.setOnClickListener { append('2') }
    three.setOnClickListener { append('3') }
    four.setOnClickListener { append('4') }
    five.setOnClickListener { append('5') }
    six.setOnClickListener { append('6') }
    seven.setOnClickListener { append('7') }
    eight.setOnClickListener { append('8') }
    nine.setOnClickListener { append('9') }
    zero.setOnClickListener { append('0') }
    dzero.setOnClickListener { append('0')
    append('0') }
    dot.setOnClickListener {
    if(dotflag == false){
    append('.')
    dotflag = true
    }
    }
    sum.setOnClickListener {
    operator()
    operator = '+'
    signView.text = operator.toString()
    }
    sub.setOnClickListener {
    operator()
    operator = '-'
    signView.text = operator.toString()
    }
    mul.setOnClickListener {
    operator()
    operator = '*'
    signView.text = operator.toString()
    }
    div.setOnClickListener {
    operator()
    operator = '/'
    signView.text = operator.toString()
    }
    cs.setOnClickListener {
    signView.text = "0"
    dotflag = false
    text = ""
    }
    ans.setOnClickListener {
    if(switch == true){
    text = signView.text.toString()
    secondnumber = text.toFloat()
    switch = false
    signView.text = ""
    when(operator)
    {
    '+' -> signView.text = (firstnumber + secondnumber).toString()
    '-' -> signView.text = (firstnumber - secondnumber).toString()
    '*' -> signView.text = (firstnumber * secondnumber).toString()
    '/' -> if(secondnumber == 0F){
    signView.text = "Error"
    }
    else
    {
    signView.text = (firstnumber / secondnumber).toString()
    }
    }
    }
    }
    cos.setOnClickListener {
    if(signView.text == ""){
    }
    else
    {
    var result:Float = 0F
    text = signView.text.toString()
    result = cos(text.toFloat()*((3.141592653589793238462643/180).toFloat()))
    signView.text = result.toString()
    }
    }
    sin.setOnClickListener {
    if(signView.text == ""){
    }
    else
    {
    var result:Float = 0F
    text = signView.text.toString()
    result = sin(text.toFloat()*((3.141592653589793238462643/180).toFloat()))
    signView.text = result.toString()
    }
    }
    tan.setOnClickListener {
    if(signView.text == ""){
    }
    else
    {
    var result:Float = 0F
    text = signView.text.toString()
    result = tan(text.toFloat()*((3.141592653589793238462643/180).toFloat()))
    signView.text = result.toString()
    }
    }
    }
    fun append(number : Char = '0'){
    if(signView.text == "Error" || signView.text == "0"){
    signView.text = ""
    text = text + number
    signView.text = text
    }
    else{
    text = signView.text.toString()
    text = text + number
    signView.text = text
    }
    }
    fun operator(){
    if(switch == false){
    text = signView.text.toString()
    firstnumber = text.toFloat()
    switch = true
    dotflag = false
    signView.text = ""
    }
    }
    }
